import cv2 # Coba ganti dengan CAP_DSHOW atau CAP_V4L2 jika perlu cap = cv2.VideoCapture(0, cv2.CAP_DSHOW) if not cap.isOpened(): print("Tidak bisa membuka kamera") exit() while True: ret, frame = cap.read() if not ret: print("Tidak bisa menangkap frame") break cv2.imshow('Frame', frame) if cv2.waitKey(1) & 0xFF == ord('q'): break cap.release() cv2.destroyAllWindows()
